function persistence(num, count = 0) {  
    if (num > 9) {
    let nums = num.toString().split('');
    let numsReduce = nums.reduce((a, b) => a * +b);
                            
    if (numsReduce > 9) { 
        return persistence(numsReduce,count + 1);
        } else {
            return count + 1;
        }
    } else {
            return count;
    }
}
                

function declareWinner(f1, f2, fa) {
    if (fa == f1.name && f1.health > 0 && f2.health > 0) {
        f2.health = f2.health - f1.damagePerAttack
        return declareWinner(f1, f2, f2.name)
    } else if (fa == f2.name && f1.health > 0 && f2.health > 0){
        f1.health = f1.health - f2.damagePerAttack
        return declareWinner(f1, f2, f1.name)
    } else if (f1.health > 0){
        return f1.name
    } else {
        return f2.name
    }
}
